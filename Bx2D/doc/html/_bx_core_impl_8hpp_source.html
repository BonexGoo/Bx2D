344  template<typename TYPE, bool FORTHREAD = true, int MAXCOUNT = 256>
345  class BxVarQueue
346  {
347  private:
348  class QueueElement
349  {
350  public:
351  TYPE* Data;
352  QueueElement* Prev;
353  QueueElement* Next;
354  QueueElement(TYPE* data) : Data(data), Prev(this), Next(this) {}
355  ~QueueElement()
356  {
357  if(Prev) Prev->Next = nullptr;
358  QueueElement* CurrNode = nullptr;
359  QueueElement* NextNode = Next;
360  while(CurrNode = NextNode)
361  {
362  NextNode = CurrNode->Next;
363  CurrNode->Prev = nullptr;
364  CurrNode->Next = nullptr;
365  delete CurrNode;
366  }
367  }
368  void Enqueue(TYPE* data)
369  {
370  QueueElement* NewNode = new QueueElement(data);
371  (NewNode->Prev = Prev)->Next = NewNode;
372  (NewNode->Next = this)->Prev = NewNode;
373  }
374  TYPE* Dequeue()
375  {
376  QueueElement* DeleteNode = Next;
377  DeleteNode->Next->Prev = DeleteNode->Prev;
378  DeleteNode->Prev->Next = DeleteNode->Next;
379  DeleteNode->Prev = nullptr;
380  DeleteNode->Next = nullptr;
381  TYPE* Result = DeleteNode->Data;
382  delete DeleteNode;
383  return Result;
384  }
385  };

387  private:
388  QueueElement Head;
389  int DataCount;
390  id_mutex Mutex;
391

402  if(FORTHREAD) BxCore::Thread::CloseMutex(Mutex);

405  public:
406  void Enqueue(TYPE* data)
407  {
408  if(FORTHREAD) BxCore::Thread::Lock(Mutex);
409  {
410  Head.Enqueue(data);
411  DataCount++;
412  BxASSERT("BxVarQueue<큐의 데이터수량이 경고수준입니다>", DataCount != MAXCOUNT);
413  }
414  if(FORTHREAD) BxCore::Thread::Unlock(Mutex);
415  }

418  {
419  TYPE* Data = nullptr;
420  if(FORTHREAD) BxCore::Thread::Lock(Mutex);
421  if(0 < DataCount)
422  {

424  DataCount--;
425  }
