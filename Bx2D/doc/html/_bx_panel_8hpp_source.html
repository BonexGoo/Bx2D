483  private: global_func bool CalcTouch(systouchtype type, int x, int y)
484  {
485  BxPanel Panel;
486  if(!Panel.Single->RefDraw) return false;
487  // Operate last panel
488  BxPanel* CurPanel = Panel.Single->LastPanel;
489  BxPanel* CurPanelScrolled = Panel.Single->LastPanelScrolled;
490  if(CurPanel)
491  {
492  if(CurPanel == CurPanelScrolled)
493  {
494  switch(type)
495  {
496  case systouchtype_move:
497  CurPanelScrolled->VarL->ForceScroll(CurPanelScrolled,
498  (x - Panel.Single->LastTouch[0].x) * ScrollSensitive,
499  (y - Panel.Single->LastTouch[0].y) * ScrollSensitive);
500  CurPanelScrolled->VarL->LastScrollPower =
501  BxUtilGlobal::Min(CurPanelScrolled->VarL->LastScrollPower + ScrollSensitive / 8, ScrollSensitive);
502  Panel.Single->LastTouch[2] = Panel.Single->LastTouch[1];
503  Panel.Single->LastTouch[1] = Panel.Single->LastTouch[0];
504  Panel.Single->LastTouch[0] = XY(x, y);
505  break;
506  case systouchtype_up:
507  case systouchtype_cancel:
508  Panel.Single->LastPanel = nullptr;
509  Panel.Single->LastPanelScrolled = nullptr;
510  CurPanelScrolled->VarL->LastScrollPhy.x = (x - Panel.Single->LastTouch[2].x) * CurPanelScrolled->VarL->LastScrollPower;
511  CurPanelScrolled->VarL->LastScrollPhy.y = (y - Panel.Single->LastTouch[2].y) * CurPanelScrolled->VarL->LastScrollPower;
512  break;
513  }
514  return true;
515  }
516  else
517  {
518  const rect& Rect = CurPanel->VarL->LastRect;
519  bool IsOutside = (x < Rect.l || Rect.r <= x || y < Rect.t || Rect.b <= y);
520  switch(type)
521  {
522  case systouchtype_move:
523  if(CurPanelScrolled)
524  {
525  Panel.Single->LastPanel = CurPanelScrolled;
526  Panel.Single->LastTouch[0] = Panel.Single->LastTouch[1] = Panel.Single->LastTouch[2] = XY(x, y);
527  CurPanel->OnTouch(tchUpAuto);
528  return true;
529  }
530  else CurPanel->OnTouch((IsOutside)? tchMoveOut : tchMoveIn);
531  break;
532  case systouchtype_down: IsOutside = false; // For key pressed
533  case systouchtype_up:
534  case systouchtype_cancel:
535  {
536  Panel.Single->LastPanel = nullptr;
537  Panel.Single->LastPanelScrolled = nullptr;
538  string Message = CurPanel->OnTouch((IsOutside)? tchUpOut : tchUpIn);
539  if(!IsOutside) CurPanel->VarL->ClickedData.Calling(CurPanel, Message);
540  }
541  break;
542  }
543  }
544  }
545  // Look into new panel
546  static const int XLength = (Panel.Single->RefDraw->Width(true) + CellSize - 1) / CellSize;
547  static const int YLength = (Panel.Single->RefDraw->Height(true) + CellSize - 1) / CellSize;
548  const int X = x / CellSize, Y = y / CellSize;
549  if(X < 0 || XLength <= X || Y < 0 || YLength <= Y) return false;
550  Singleton::PanelCell& Cell = Panel.Single->PanelCells[X + Y * XLength];
551  if(Cell.WriteID != Panel.Single->LastWriteID) return false;
552  for(int i = Cell.Length - 1; 0 <= i; --i)
553  {
554  const rect& Rect = Cell.Stack[i]->VarL->LastRect;
555  if(x < Rect.l || Rect.r <= x || y < Rect.t || Rect.b <= y)
556  continue;
557  if(CurPanel != Cell.Stack[i])
558  {
559  CurPanel = Cell.Stack[i];
560  CurPanelScrolled = nullptr;
561  switch(type)
562  {
563  case systouchtype_down:
564  // Search scrolled panel at under new panel
565  while(0 <= --i)
566  {
567  VariableL* CurL = (VariableL*) Cell.Stack[i]->VarL;
568  const rect& Rect = CurL->LastRect;
569  if(x < Rect.l || Rect.r <= x || y < Rect.t || Rect.b <= y)
570  continue;
571  if(CurL->LastType == viewScroll)
572  {
573  CurPanelScrolled = Cell.Stack[i];
574  break;
575  }
576  }
577  Panel.Single->LastPanel = CurPanel;
578  Panel.Single->LastPanelScrolled = CurPanelScrolled;
579  CurPanel->OnTouch(tchDown);
580  CurPanel->VarL->ForceScroll(CurPanel, 0, 0);
581  break;
582  case systouchtype_move:
583  CurPanel->OnTouch(tchDrag);
584  break;
585  case systouchtype_up:
586  case systouchtype_cancel:
587  CurPanel->OnTouch(tchDrop);
588  break;
589  }
590  }
591  return true;
592  }
593  return (CurPanel != nullptr);
594  }
