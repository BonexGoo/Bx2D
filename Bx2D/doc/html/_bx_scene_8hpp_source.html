414  global_func bool __OnEvent__(BxDraw& Draw)
415  {
416  bool DoEvent = false;
417  EventQueue* TopNode = EventQueue::Top(true);
418  EventQueue* CurNode = TopNode;
419  while(CurNode = CurNode->Next)
420  {
421  DoEvent = true;
422  sysevent& Event = CurNode->Event;
423  syseventset Set = CurNode->Set;
424  bool NeedRecall = true;
425  while(NeedRecall)
426  {
427  NeedRecall = false;
428  switch(Event.type)
429  {
430  case syseventtype_touch:
431  //if(Event.touch.id == 0)
432  {
433  int SelectedButton = -1;
434  if(Event.touch.type == systouchtype_down)
435  {
436  for(int i = Draw.ButtonLength - 1; 0 <= i; --i)
437  {
438  if(Event.touch.x < Draw.ButtonEvent[i].Field.l || Event.touch.y < Draw.ButtonEvent[i].Field.t ||
439  Draw.ButtonEvent[i].Field.r <= Event.touch.x || Draw.ButtonEvent[i].Field.b <= Event.touch.y)
440  continue;
441  SelectedButton = i;
442  Draw.CapturedButtonEvent.SetEvent(Draw.ButtonEvent[i].Name, Draw.ButtonEvent[i].Field);
443  break;
444  }
445  }
446  else if(*Draw.CapturedButtonEvent.Name)
447  {
448  for(int i = Draw.ButtonLength - 1; 0 <= i; --i)
449  {
450  if(BxUtilGlobal::StrCmp(Draw.CapturedButtonEvent.Name, Draw.ButtonEvent[i].Name) != same)
451  continue;
452  SelectedButton = i;
453  break;
454  }
455  if(Event.touch.type != systouchtype_move)
456  {
457  Draw.CapturedButtonEvent.DeleteEvent();
458  NeedRecall = true;
459  }
460  }
461  if(-1 < SelectedButton && Event.touch.kind <= inputkind_mouse_left)
462  {
463  Event.type = syseventtype_button;
464  switch(Event.touch.type)
465  {
466  case systouchtype_down: Event.button.type = sysbuttontype_down; break;
467  case systouchtype_move: Event.button.type = sysbuttontype_drag; break;
468  case systouchtype_up:
469  case systouchtype_cancel: Event.button.type = sysbuttontype_up; break;
470  }
471  Event.button.name = Draw.ButtonEvent[SelectedButton].Name;
472  Event.button.field = Draw.ButtonEvent[SelectedButton].Field;
473  Event.button.x = Event.touch.x - Draw.ButtonEvent[SelectedButton].Field.l;
474  Event.button.y = Event.touch.y - Draw.ButtonEvent[SelectedButton].Field.t;
475  }
476  }
477  break;
478  case syseventtype_button:
479  if(Event.button.type == sysbuttontype_up)
480  {
481  for(int i = Draw.ButtonLength - 1; 0 <= i; --i)
482  {
483  if(Event.touch.x < Draw.ButtonEvent[i].Field.l || Event.touch.y < Draw.ButtonEvent[i].Field.t ||
484  Draw.ButtonEvent[i].Field.r <= Event.touch.x || Draw.ButtonEvent[i].Field.b <= Event.touch.y)
485  continue;
486  if(BxUtilGlobal::StrCmp(Event.button.name, Draw.ButtonEvent[i].Name) != same)
487  {
488  Event.button.type = sysbuttontype_drop;
489  Event.button.name = Draw.ButtonEvent[i].Name;
490  Event.button.field = Draw.ButtonEvent[i].Field;
491  Event.button.x = Event.touch.x - Draw.ButtonEvent[i].Field.l;
492  Event.button.y = Event.touch.y - Draw.ButtonEvent[i].Field.t;
493  }
494  break;
495  }
496  if(Event.button.type != sysbuttontype_drop)
497  {
498  NeedRecall = false;
499  continue;
500  }
501  }
502  break;
503  }
504  // 이벤트처리
505  ActiveScene* OneActiveScene = &__GetActiveScene__();
506  bool IsEventDone = (Event.type == syseventtype_touch && Event.touch.id == 0
507  && BxPanel::CalcTouch(Event.touch.type, Event.touch.x, Event.touch.y));
508  while(!IsEventDone && (OneActiveScene = OneActiveScene->Next) != nullptr)
509  {
510  if(OneActiveScene->IsFlowing())
511  continue;
512  int OldTouchX = 0, OldTouchY = 0, FlowedX = 0, FlowedY = 0;
513  switch(Event.type)
514  {
515  case syseventtype_init: break;
516  case syseventtype_quit: break;
517  case syseventtype_resize: break;
518  case syseventtype_key:
519  if(Set == syseventset_need_enable && OneActiveScene->EnableKey == false) continue;
520  if(Set == syseventset_do_enable) OneActiveScene->EnableKey = true;
521  break;
522  case syseventtype_touch:
523  OldTouchX = Event.touch.x;
524  OldTouchY = Event.touch.y;
525  OneActiveScene->ToFlow(FlowedX, FlowedY);
526  Event.touch.x -= FlowedX;
527  Event.touch.y -= FlowedY;
528  case syseventtype_button:
529  if(Set == syseventset_need_enable && OneActiveScene->EnableTouch == false) continue;
530  if(Set == syseventset_do_enable) OneActiveScene->EnableTouch = true;
531  break;
532  }
533  if(OneActiveScene->Scene->Event(OneActiveScene->Scene->GetData(sysmethod_get), Event) == syseventresult_done)
534  {
535  BxASSERT("BxScene<해당 이벤트는 독점할 수 없습니다>", Event.type != syseventtype_init);
536  BxASSERT("BxScene<해당 이벤트는 독점할 수 없습니다>", Event.type != syseventtype_quit);
537  BxASSERT("BxScene<해당 이벤트는 독점할 수 없습니다>", Event.type != syseventtype_resize);
538  IsEventDone = true;
539  }
540  if(Event.type == syseventtype_touch)
541  {
542  Event.touch.x = OldTouchX;
543  Event.touch.y = OldTouchY;
544  }
545  }
546  // 이벤트완료시 하위처리
547  if(IsEventDone)
548  while((OneActiveScene = OneActiveScene->Next) != nullptr)
549  {
550  switch(Event.type)
551  {
552  case syseventtype_init: break;
553  case syseventtype_quit: break;
554  case syseventtype_resize: break;
555  case syseventtype_key:
556  OneActiveScene->EnableKey = false;
557  break;
558  case syseventtype_touch:
559  case syseventtype_button:
560  OneActiveScene->EnableTouch = false;
561  break;
562  }
563  }
564  __UpdateActiveScene__();
565  }
566  }
567  delete TopNode->Next;
568  TopNode->Next = nullptr;
569  return DoEvent;
570  }

574  global_func bool __OnEventUpdateRender__(BxDraw& Draw)
575  {
576  // Event
577  sysupdateresult UpdateResult = (BxScene::__OnEvent__(Draw))? sysupdateresult_do_render : sysupdateresult_no_render_fixed_frame;
578  // Update
579  ActiveScene* OneActiveScene = &__GetActiveScene__();
580  while((OneActiveScene = OneActiveScene->Next) != nullptr)
581  {
582  if(OneActiveScene->IsFlowing())
583  {
584  UpdateResult = (sysupdateresult) (UpdateResult | sysupdateresult_do_render_fixed_frame);
585  continue;

587  UpdateResult = (sysupdateresult) (UpdateResult | OneActiveScene->Scene->Update(OneActiveScene->Scene->GetData(sysmethod_get)));
588  }
589  // Render
590  const bool DoRender = ((UpdateResult & sysupdateresult_do_render_fixed_frame) == sysupdateresult_do_render_fixed_frame);
591  if(DoRender)
592  {
593  BxPanel::PaintReady(Draw);
594  if((UpdateResult & sysupdateresult_do_render) == sysupdateresult_do_render)
595  BxCore::Main::PullFrameDelay();
596  const int ActiveSceneCount = __GetActiveSceneCount__();
597  for(int i = ActiveSceneCount; 0 < i; --i)
598  {
599  // 역순접근
600  ActiveScene* OneActiveScene = &__GetActiveScene__();
601  for(int j = 0; j < i; ++j) OneActiveScene = OneActiveScene->Next;
602  // 마진모드 설정
603  const bool IsGeneralScene = (__GetActiveSceneGUICount__() < i);
604  const bool OldMarginMode = BxCore::Surface::SetMarginMode(IsGeneralScene);
605  // 클리핑 스타일링 설정
606  int FlowedX = (IsGeneralScene)? BxCore::Main::GetGUIMarginL() : 0;
607  int FlowedY = (IsGeneralScene)? BxCore::Main::GetGUIMarginT() : 0;
608  OneActiveScene->ToFlow(FlowedX, FlowedY);
609  Draw.AddStyle(CLIP(XYWH(FlowedX, FlowedY, BxCore::Surface::GetWidthHW(), BxCore::Surface::GetHeightHW())));
610  // 랜더링호출
611  OneActiveScene->Scene->Render(OneActiveScene->Scene->GetData(sysmethod_get), Draw);
612  // 전체 스타일링 해제
613  Draw.SubStyle(true);
614  // 마진모드 해제
615  BxCore::Surface::SetMarginMode(OldMarginMode);
616  }
617  // 씬이동
618  ActiveScene* OneActiveScene = &__GetActiveScene__();
619  ActiveScene* NextActiveScene = OneActiveScene->Next;
620  while((OneActiveScene = NextActiveScene) != nullptr)
621  {
622  NextActiveScene = OneActiveScene->Next;
623  OneActiveScene->DoFlow();
624  }
625  }
626  else if((UpdateResult & sysupdateresult_no_render) == sysupdateresult_no_render)
627  BxCore::Main::PushFrameDelay();
628  __UpdateActiveScene__();
629  return DoRender;
630  }
