230  void touchEvent(QTouchEvent* event)
231  {
232  if(SavedTouchFlag == 0)
233  {
234  PostMouseEventsByWheelDone();
235  TouchPressed = true;
236  }
237  uint CurTouchFlag = 0;
238  QList<QTouchEvent::TouchPoint> Points = event->touchPoints();
239  foreach(const QTouchEvent::TouchPoint& Point, Points)
240  {
241  if(Point.state() == Qt::TouchPointStationary)
242  continue;
243  const int ID = Point.id();
244  if(31 < ID) continue;
245  const int X = (int) Point.pos().x();
246  const int Y = (int) Point.pos().y();
247  const bool IsPressed = ((SavedTouchFlag & (1 << ID)) != 0);
248  PostMouseEvent((IsPressed)? systouchtype_move : systouchtype_down, ID, X, Y);
249  CurTouchFlag |= 1 << ID;
250  SavedTouchPos[ID] = BxDrawGlobal::XY(X, Y);
251  }
252  for(int i = 0; i < 32; ++i)
253  {
254  const bool IsPressed = ((SavedTouchFlag & (1 << i)) != 0);
255  const bool IsPressNow = ((CurTouchFlag & (1 << i)) != 0);
256  if(IsPressed && !IsPressNow)
257  PostMouseEvent(systouchtype_up, i, SavedTouchPos[i].x, SavedTouchPos[i].y);
258  }
259  SavedTouchFlag = CurTouchFlag;
260  }
261  void touchEventDone()
262  {
263  TouchPressed = false;

265  if((SavedTouchFlag & (1 << i)) != 0)
266  PostMouseEvent(systouchtype_up, i, SavedTouchPos[i].x, SavedTouchPos[i].y);
267  SavedTouchFlag = 0;
